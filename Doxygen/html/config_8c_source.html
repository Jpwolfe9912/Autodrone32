49void parseRcChannels(const char *input)
50{
51 const char *c, *s;
52
53 for (c = input; *c; c++)
54 {
55 s = strchr(rcChannelLetters, *c);
56 if (s)
57 eepromConfig.rcMap[s - rcChannelLetters] = c - input;
58 }
59}

72void readEEPROM(void)
73{
74 eepromConfig_t *dst = &eepromConfig;
75
76 *dst = *(eepromConfig_t*)FLASH_WRITE_EEPROM_ADDR ;
77
78 if ( crcCheckVal != crc32bEEPROM(dst, true) )
79 {
80 evrPush(EVR_FlashCRCFail,0);
81 dst->CRCFlags |= CRC_HistoryBad;
82 }
83 else if ( dst->CRCFlags & CRC_HistoryBad )
84 evrPush(EVR_ConfigBadHistory,0);
85
86 accConfidenceDecay = 1.0f / sqrt(eepromConfig.accelCutoff);
87
88 eepromConfig.yawDirection = constrain(eepromConfig.yawDirection, -1.0f, 1.0f);
89}

93int writeEEPROM(void)
94{
95 FLASH_Status status;
96 int32_t i;
97
98 eepromConfig_t *src = &eepromConfig;
99 uint32_t *dst = (uint32_t*)FLASH_WRITE_EEPROM_ADDR;
100
101 // there's no reason to write these values to EEPROM, they'll just be noise
102 zeroPIDstates();
103
104 if ( src->CRCFlags & CRC_HistoryBad )
105 evrPush(EVR_ConfigBadHistory,0);
106
107 src->CRCAtEnd[0] = crc32B( (uint32_t*)&src[0], src->CRCAtEnd);
108
109 FLASH_Unlock();
110
111 FLASH_ClearFlag(FLASH_FLAG_EOP | FLASH_FLAG_OPERR | FLASH_FLAG_WRPERR |
112 FLASH_FLAG_PGAERR | FLASH_FLAG_PGPERR | FLASH_FLAG_PGSERR);
113
114 status = FLASH_EraseSector(FLASH_Sector_1, VoltageRange_3);
115
117
118 i = -1;
119
120 while ( FLASH_COMPLETE == status && i++ < eepromConfigNUMWORD )
121 status = FLASH_ProgramWord((uint32_t)&dst[i], ((uint32_t*)src)[i]);
122
123 if ( FLASH_COMPLETE != status )
124 evrPush( -1 == i ? EVR_FlashEraseFail : EVR_FlashProgramFail, status);
125
127
128 FLASH_Lock();
129
130 readEEPROM();
131
132 return status;
133}

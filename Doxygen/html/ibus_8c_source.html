113static ibusStatus_e
114ibus_process_frame(void)
115{
116 uint8_t b = rxBuf[framePos++];
117 status = IBUS_BUSY;
118 switch(state){
119 case IBUS_STATE_LENGTH: { // length byte
120 if(b == 0x20){
121 frameLength = b;
122 state++;
123 }
124 else
125 status = IBUS_ERROR;
126 break;
127 }
128 case IBUS_STATE_TYPE: { // type byte
129 if(b == 0x40){
130 devID = b;
131 state++;
132 }
133 else
134 status = IBUS_ERROR;
135 break;
136 }
137 case IBUS_STATE_PAYLOAD: { // payload bytes
138 ibusPayload[payloadPos++] = b;
139 if(payloadPos == PAYLOAD_SIZE)
140 {
141 ibus_update(ibusPayload);
142 state++;
143 payloadPos = 0;
144 }
145 break;
146 }
147 case IBUS_STATE_CRC: { // crc bytes
148 ibusCRC[crcPos++] = b;
149 if(crcPos == CRC_SIZE){
150 if(ibus_frame_crc(ibusCRC))
151 {
152 status = IBUS_READY;
153 }
154 else{
155 status = IBUS_ERROR;
156 }
157 crcPos = 0;
158 }
159 break;
160 }
161 }
162 if(status == IBUS_ERROR)
163 {
164 ibus_error_timeout();
165
166 state = 0;
167 }
168 if(status == IBUS_READY)
169 {
170 state = 0;
171 }
172 return status;
173}

244static void
245ibus_update(uint8_t *pData)
246{
247 for(uint8_t ch_index = 0, bf_index = 0; ch_index < RC_CHANNELS; ch_index++, bf_index += 2)
248 {
249 ibusChannels[ch_index] = pData[bf_index + 1] << 8 | pData[bf_index];
250 }
251}

262static bool
263ibus_frame_crc(uint8_t *pData)
264{
265 uint16_t checksum_cal = 0xFFFF;
266 uint16_t checksum_ibus;
267
268 for(uint8_t i = 0; i < 30; i++)
269 {
270 checksum_cal -= rxBuf[i];
271 }
272
273 checksum_ibus = pData[1] << 8 | pData[0]; // checksum value from ibus
274 return (checksum_ibus == checksum_cal);
275}

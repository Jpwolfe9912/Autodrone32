109ibus_process_frame(void)
110{
111 uint8_t b = rxBuf[framePos++];
112
113 status = IBUS_BUSY;
114 switch(state){
115 case IBUS_STATE_LENGTH: { // length byte
116 if(b == 0x20){
117 errorTimeOut = 0;
118
119 frameLength = b;
120 state++;
121 }
122 else
123 {
124 errorTimeOut++;
125 status = IBUS_ERROR;
126 }
127 break;
128 }
129 case IBUS_STATE_TYPE: { // type byte
130 if(b == 0x40){
131 devID = b;
132 state++;
133 }
134 else
135 status = IBUS_ERROR;
136 break;
137 }
138 case IBUS_STATE_PAYLOAD: { // payload bytes
139 ibusPayload[payloadPos++] = b;
140 if(payloadPos == PAYLOAD_SIZE)
141 {
142 ibus_update(ibusPayload);
143 state++;
144 payloadPos = 0;
145 }
146 break;
147 }
148 case IBUS_STATE_CRC: { // crc bytes
149 ibusCRC[crcPos++] = b;
150 if(crcPos == CRC_SIZE){
151 if(ibus_frame_crc(ibusCRC))
152 {
153 status = IBUS_READY;
154 }
155 else{
156 status = IBUS_ERROR;
157 }
158 crcPos = 0;
159 }
160 break;
161 }
162 }
163 ibus_timeout();
164
165 if(status == IBUS_ERROR)
166 {
167 framePos = 0;
168 state = 0;
169 }
170 if(status == IBUS_READY)
171 {
172 readyTimeOut++;
173
174 framePos = 0;
175 state = 0;
176 }
177
178 return status;

206static bool
207ibus_frame_crc(uint8_t *pData)
208{
209 uint16_t checksum_cal = 0xFFFF;
210 uint16_t checksum_ibus;
211
212 for(uint8_t i = 0; i < 30; i++)
213 {
214 checksum_cal -= rxBuf[i];
215 }
216
217 checksum_ibus = pData[1] << 8 | pData[0]; // checksum value from ibus
218
219 if(prevCRC == checksum_ibus)
220 failsafeTimeOut++;
221 else
222 failsafeTimeOut = 0;
223
224 prevCRC = checksum_ibus;
225
226 return (checksum_ibus == checksum_cal);

69void processFlightCommands(void)
70{
71 uint8_t channel;
72 uint8_t channelsToRead = 8;
73
74 if ( rcActive == true )
75 {
76
77 /* Makes RPY from -1000 to 1000 */
78 rxCommands[ROLL] = (ibusChannels[ROLL] * 2) - MIDCOMMAND; // Roll Range -1000:1000
79 rxCommands[PITCH] = (ibusChannels[PITCH] * 2) - MIDCOMMAND; // Pitch Range -1000:1000
80 rxCommands[YAW] = (ibusChannels[YAW] * 2) - MIDCOMMAND; // Yaw Range -1000:1000
81
82 /* Makes all other channels from 2000 to 4000 */
83 for (channel = 3; channel < channelsToRead; channel++)
84 rxCommands[channel] = ibusChannels[channel] * 2;
85 }
86
87 // Set past command in detent values
88 for (channel = 0; channel < 3; channel++)
89 previousCommandInDetent[channel] = commandInDetent[channel];
90
91 // Apply deadbands and set detent discretes'
92 for (channel = 0; channel < 3; channel++)
93 {
94 /* RPY is within deadband */
95 if ((rxCommands[channel] <= DEADBAND) && (rxCommands[channel] >= -DEADBAND))
96 {
97 rxCommands[channel] = 0; // set command to 0
98 commandInDetent[channel] = true; // command IS within detent
99 }
100 else
101 {
102 commandInDetent[channel] = false;
103 if (rxCommands[channel] > 0)
104 {
105 rxCommands[channel] = (rxCommands[channel] - DEADBAND) * DEADBAND_SLOPE;
106 }
107 else
108 {
109 rxCommands[channel] = (rxCommands[channel] + DEADBAND) * DEADBAND_SLOPE;
110 }
111 }
112 }
113
115
116 /* Check for disarm switch */
117 if(rxCommands[AUX1] < MIDCOMMAND){
118 zeroPIDstates();
119 armed = false;
120 disarmingTimer = 0;
121
122 // Check for gyro bias command ( low throttle, left yaw, aft pitch, right roll )
123 if ( (rxCommands[YAW ] < (eepromConfig.minCheck - MIDCOMMAND)) && //mincheck = 2200
124 (rxCommands[ROLL ] > (eepromConfig.maxCheck - MIDCOMMAND)) && //maxcheck = 3800
125 (rxCommands[PITCH] < (eepromConfig.minCheck - MIDCOMMAND)) )
126 {
127 computeMPU6000RTData();
128// pulseMotors(3);
129 }
130 }
131
132 /* Check for arm switch and throttle low(<2200) */
133 if((rxCommands[AUX1] > MIDCOMMAND) &&
134 (rxCommands[THROTTLE] < eepromConfig.minCheck) &&
135 (armed == false)){
136 zeroPIDstates();
137 armed = true;
138 }
139
140
142
143 // Check for armed true and throttle command > minThrottle
144
145 if ((armed == true) && (rxCommands[THROTTLE] > eepromConfig.minThrottle))
146 pidReset = false;
147 else
148 pidReset = true;
149
151
152 // Check AUX1 for rate, attitude, or GPS mode (3 Position Switch) NOT COMPLETE YET....
153
154 if ((rxCommands[AUX2] > MIDCOMMAND) &&
155 (mode == ROVER) &&
156 (armed == false))
157 {
158 mode = FLIGHT;
159 setPIDstates(ROLL_ATT_PID, 0.0f);
160 setPIDstates(PITCH_ATT_PID, 0.0f);
161 setPIDstates(HEADING_PID, 0.0f);
162 }
163 else if ((rxCommands[AUX2] <= MIDCOMMAND) &&
164 (mode == FLIGHT) &&
165 (armed == false))
166 {
167 mode = ROVER;
168 }
169
171}

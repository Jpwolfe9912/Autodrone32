81// Read Flight Commands
83
84void processFlightCommands(void)
85{
86 uint8_t channel;
87 uint8_t channelsToRead = 8;
88
89 float hdgDelta, simpleX, simpleY;
90
91 if ( rcActive == true )
92 {
93// // Read receiver commands
94// if (eepromConfig.receiverType == PPM)
95// channelsToRead = eepromConfig.ppmChannels;
96//
97// for (channel = 0; channel < channelsToRead; channel++)
98// {
99// if (eepromConfig.receiverType == SPEKTRUM)
100// ibusChannels[channel] = spektrumRead(eepromConfig.rcMap[channel]);
101// else if (eepromConfig.receiverType == SBUS)
102// ibusChannels[channel] = sBusRead(eepromConfig.rcMap[channel]);
103// else
104// ibusChannels[channel] = rxRead(eepromConfig.rcMap[channel]);
105// }
106
107// ibusChannels[ROLL] -= eepromConfig.midCommand; // Roll Range -1000:1000
108// ibusChannels[PITCH] -= eepromConfig.midCommand; // Pitch Range -1000:1000
109// ibusChannels[YAW] -= eepromConfig.midCommand; // Yaw Range -1000:1000
110
111 ibusChannels[ROLL] = (ibusChannels[ROLL] * 2) - 2000; // Roll Range -1000:1000
112 ibusChannels[PITCH] = (ibusChannels[PITCH] * 2) - 2000; // Pitch Range -1000:1000
113 ibusChannels[YAW] = (ibusChannels[YAW] * 2) - 2000; // Yaw Range -1000:1000
114
115// for (channel = 3; channel < channelsToRead; channel++)
116// ibusChannels[channel] -= eepromConfig.midCommand - MIDCOMMAND; // Range 2000:4000
117 for (channel = 3; channel < channelsToRead; channel++)
118 ibusChannels[channel] *= 2;
119 }
120
121 // Set past command in detent values
122 for (channel = 0; channel < 3; channel++)
123 previousCommandInDetent[channel] = commandInDetent[channel];
124
125 // Apply deadbands and set detent discretes'
126 for (channel = 0; channel < 3; channel++)
127 {
128 if ((ibusChannels[channel] <= DEADBAND) && (ibusChannels[channel] >= -DEADBAND))
129 {
130 ibusChannels[channel] = 0;
131 commandInDetent[channel] = true;
132 }
133 else
134 {
135 commandInDetent[channel] = false;
136 if (ibusChannels[channel] > 0)
137 {
138 ibusChannels[channel] = (ibusChannels[channel] - DEADBAND) * DEADBAND_SLOPE;
139 }
140 else
141 {
142 ibusChannels[channel] = (ibusChannels[channel] + DEADBAND) * DEADBAND_SLOPE;
143 }
144 }
145 }
146
148
149 /* Check for disarm switch */
150 if(ibusChannels[AUX1] < MIDCOMMAND){
151 zeroPIDstates();
152 armed = false;
153 disarmingTimer = 0;
154 }
155 /* Check for arm switch and throttle low */
156 if((ibusChannels[AUX1] > MIDCOMMAND) && (ibusChannels[THROTTLE] < eepromConfig.minCheck) && (armed == false)){
157 zeroPIDstates();
158 armed = true;
159 }
160
161 // Check for low throttle
162 if ( ibusChannels[THROTTLE] < eepromConfig.minCheck )
163 {
164 // Check for disarm command ( low throttle, left yaw )
165 if ( (ibusChannels[YAW] < (eepromConfig.minCheck - MIDCOMMAND)) && (armed == true) )
166 {
167 disarmingTimer++;
168
169 if (disarmingTimer > eepromConfig.disarmCount)
170 {
171 zeroPIDstates();
172 armed = false;
173 disarmingTimer = 0;
174 }
175 }
176 else
177 {
178 disarmingTimer = 0;
179 }
180
181 // Check for gyro bias command ( low throttle, left yaw, aft pitch, right roll )
182 if ( (ibusChannels[YAW ] < (eepromConfig.minCheck - MIDCOMMAND)) &&
183 (ibusChannels[ROLL ] > (eepromConfig.maxCheck - MIDCOMMAND)) &&
184 (ibusChannels[PITCH] < (eepromConfig.minCheck - MIDCOMMAND)) )
185 {
186 computeMPU6000RTData();
187 pulseMotors(3);
188 }
189
190 // Check for arm command ( low throttle, right yaw)
191 if ((ibusChannels[YAW] > (eepromConfig.maxCheck - MIDCOMMAND) ) && (armed == false) && (execUp == true))
192 {
193 armingTimer++;
194
195 if (armingTimer > eepromConfig.armCount)
196 {
197 zeroPIDstates();
198 armed = true;
199 armingTimer = 0;
200 }
201 }
202 else
203 {
204 armingTimer = 0;
205 }
206 }
207
209
210 // Check for armed true and throttle command > minThrottle
211
212 if ((armed == true) && (ibusChannels[THROTTLE] > eepromConfig.minThrottle))
213 pidReset = false;
214 else
215 pidReset = true;
216
218
219 // Check AUX1 for rate, attitude, or GPS mode (3 Position Switch) NOT COMPLETE YET....
220
221 if ((ibusChannels[AUX1] > MIDCOMMAND) && (flightMode == RATE))
222 {
223 flightMode = ATTITUDE;
224 setPIDstates(ROLL_ATT_PID, 0.0f);
225 setPIDstates(PITCH_ATT_PID, 0.0f);
226 setPIDstates(HEADING_PID, 0.0f);
227 }
228 else if ((ibusChannels[AUX1] <= MIDCOMMAND) && (flightMode == ATTITUDE))
229 {
230 flightMode = RATE;
231 setPIDstates(ROLL_RATE_PID, 0.0f);
232 setPIDstates(PITCH_RATE_PID, 0.0f);
233 setPIDstates(YAW_RATE_PID, 0.0f);
234 }
235
237
238 // Check yaw in detent and flight mode to determine hdg hold engaged state
239
240 if ((commandInDetent[YAW] == true) && (flightMode == ATTITUDE) && (headingHoldEngaged == false))
241 {
242 headingHoldEngaged = true;
243 setPIDstates(HEADING_PID, 0.0f);
244 setPIDstates(YAW_RATE_PID, 0.0f);
245 headingReference = heading.mag;
246 }
247
248 if (((commandInDetent[YAW] == false) || (flightMode != ATTITUDE)) && (headingHoldEngaged == true))
249 {
250 headingHoldEngaged = false;
251 }
252
254
255 // Simple Mode Command Processing
256
257 if (ibusChannels[AUX3] > MIDCOMMAND)
258 {
259 hdgDelta = sensors.attitude500Hz[YAW] - homeData.magHeading;
260
261 hdgDelta = standardRadianFormat(hdgDelta);
262
263 simpleX = cosf(hdgDelta) * ibusChannels[PITCH] + sinf(hdgDelta) * ibusChannels[ROLL ];
264
265 simpleY = cosf(hdgDelta) * ibusChannels[ROLL ] - sinf(hdgDelta) * ibusChannels[PITCH];
266
267 ibusChannels[ROLL ] = simpleY;
268
269 ibusChannels[PITCH] = simpleX;
270 }
271
273
274 // Vertical Mode Command Processing
275
276 verticalReferenceCommand = ibusChannels[THROTTLE] - eepromConfig.midCommand;
277
278 // Set past altitude reference in detent value
279 previousVertRefCmdInDetent = vertRefCmdInDetent;
280
281 // Apply deadband and set detent discrete'
282 if ((verticalReferenceCommand <= ALT_DEADBAND) && (verticalReferenceCommand >= -ALT_DEADBAND))
283 {
284 verticalReferenceCommand = 0;
285 vertRefCmdInDetent = true;
286 }
287 else
288 {
289 vertRefCmdInDetent = false;
290 if (verticalReferenceCommand > 0)
291 {
292 verticalReferenceCommand = (verticalReferenceCommand - ALT_DEADBAND) * ALT_DEADBAND_SLOPE;
293 }
294 else
295 {
296 verticalReferenceCommand = (verticalReferenceCommand + ALT_DEADBAND) * ALT_DEADBAND_SLOPE;
297 }
298 }
299
301
302 // Vertical Mode State Machine
303
304 switch (verticalModeState)
305 {
306 case ALT_DISENGAGED_THROTTLE_ACTIVE:
307 if ((ibusChannels[AUX2] > MIDCOMMAND) && (previousAUX2State <= MIDCOMMAND)) // AUX2 Rising edge detection
308 {
309 verticalModeState = ALT_HOLD_FIXED_AT_ENGAGEMENT_ALT;
310 setPIDstates(HDOT_PID, 0.0f);
311 setPIDstates(H_PID, 0.0f);
312 altitudeHoldReference = hEstimate;
313 throttleReference = ibusChannels[THROTTLE];
314 }
315
316 break;
317
319
320 case ALT_HOLD_FIXED_AT_ENGAGEMENT_ALT:
321 if ((vertRefCmdInDetent == true) || eepromConfig.verticalVelocityHoldOnly)
322 verticalModeState = ALT_HOLD_AT_REFERENCE_ALTITUDE;
323
324 if ((ibusChannels[AUX2] <= MIDCOMMAND) && (previousAUX2State > MIDCOMMAND)) // AUX2 Falling edge detection
325 verticalModeState = ALT_DISENGAGED_THROTTLE_INACTIVE;
326
327 if ((ibusChannels[AUX4] > MIDCOMMAND) && (previousAUX4State <= MIDCOMMAND)) // AUX4 Rising edge detection
328 verticalModeState = ALT_DISENGAGED_THROTTLE_ACTIVE;
329
330 break;
331
333
334 case ALT_HOLD_AT_REFERENCE_ALTITUDE:
335 if ((vertRefCmdInDetent == false) || eepromConfig.verticalVelocityHoldOnly)
336 verticalModeState = VERTICAL_VELOCITY_HOLD_AT_REFERENCE_VELOCITY;
337
338 if ((ibusChannels[AUX2] <= MIDCOMMAND) && (previousAUX2State > MIDCOMMAND)) // AUX2 Falling edge detection
339 verticalModeState = ALT_DISENGAGED_THROTTLE_INACTIVE;
340
341 if ((ibusChannels[AUX4] > MIDCOMMAND) && (previousAUX4State <= MIDCOMMAND)) // AUX4 Rising edge detection
342 verticalModeState = ALT_DISENGAGED_THROTTLE_ACTIVE;
343
344 break;
345
347
348 case VERTICAL_VELOCITY_HOLD_AT_REFERENCE_VELOCITY:
349 if ((vertRefCmdInDetent == true) && !eepromConfig.verticalVelocityHoldOnly)
350 {
351 verticalModeState = ALT_HOLD_AT_REFERENCE_ALTITUDE;
352 altitudeHoldReference = hEstimate;
353 }
354
355 if ((ibusChannels[AUX2] <= MIDCOMMAND) && (previousAUX2State > MIDCOMMAND)) // AUX2 Falling edge detection
356 {
357 verticalModeState = ALT_DISENGAGED_THROTTLE_INACTIVE;
358 altitudeHoldReference = hEstimate;
359 }
360
361
362 if ((ibusChannels[AUX4] > MIDCOMMAND) && (previousAUX4State <= MIDCOMMAND)) // AUX4 Rising edge detection
363 verticalModeState = ALT_DISENGAGED_THROTTLE_ACTIVE;
364
365 break;
366
368
369 case ALT_DISENGAGED_THROTTLE_INACTIVE:
370 if (((ibusChannels[THROTTLE] < throttleCmd + THROTTLE_WINDOW) && (ibusChannels[THROTTLE] > throttleCmd - THROTTLE_WINDOW)) ||
371 eepromConfig.verticalVelocityHoldOnly)
372 verticalModeState = ALT_DISENGAGED_THROTTLE_ACTIVE;
373
374 if ((ibusChannels[AUX2] > MIDCOMMAND) && (previousAUX2State <= MIDCOMMAND)) // AUX2 Rising edge detection
375 verticalModeState = ALT_HOLD_FIXED_AT_ENGAGEMENT_ALT;
376
377 if ((ibusChannels[AUX4] > MIDCOMMAND) && (previousAUX4State <= MIDCOMMAND)) // AUX4 Rising edge detection
378 verticalModeState = ALT_DISENGAGED_THROTTLE_ACTIVE;
379
380 break;
381 }
382
383 previousAUX2State = ibusChannels[AUX2];
384 previousAUX4State = ibusChannels[AUX4];
385
387}

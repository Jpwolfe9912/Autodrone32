70 q15_t *pState = S->pState; /* State pointer */

72 q15_t *pStateCurnt; /* Points to the current sample of the state */
73 q15_t *px, *pb; /* Temporary pointers for state and coefficient buffers */
74 q15_t mu = S->mu; /* Adaptive factor */
75 uint32_t numTaps = S->numTaps; /* Number of filter coefficients in the filter */
76 uint32_t tapCnt, blkCnt; /* Loop counters */
77 q63_t acc; /* Accumulator */
78 q15_t e = 0; /* Error of data sample */
79 q15_t alpha; /* Intermediate constant for taps update */
80 q31_t coef; /* Temporary variable for coefficient */
81 q31_t acc_l, acc_h; /* Temporary input */
82 int32_t lShift = (15 - (int32_t) S->postShift); /* Post shift */
83 int32_t uShift = (32 - lShift);
84
85 /* S->pState points to buffer which contains previous frame (numTaps - 1) samples */
86 /* pStateCurnt points to the location where the new input data should be written */
87 pStateCurnt = &(S->pState[(numTaps - 1U)]);
88
89 /* initialise loop count */
90 blkCnt = blockSize;

93 {
94 /* Copy the new input sample into the state buffer */
95 *pStateCurnt++ = *pSrc++;
96
97 /* Initialize pState pointer */
98 px = pState;
99
100 /* Initialize coefficient pointer */
101 pb = pCoeffs;
102
103 /* Set the accumulator to zero */
104 acc = 0;
105
106#if defined (ARM_MATH_LOOPUNROLL)
107
108 /* Loop unrolling: Compute 4 taps at a time. */
109 tapCnt = numTaps >> 2U;
110
111 while (tapCnt > 0U)

113 /* Perform the multiply-accumulate */
114 /* acc += b[N] * x[n-N] + b[N-1] * x[n-N-1] */

116 acc = __SMLALD(read_q15x2_ia (&px), read_q15x2_ia (&pb), acc);
117
118 /* Decrement loop counter */
119 tapCnt--;
120 }
121
122 /* Loop unrolling: Compute remaining taps */
123 tapCnt = numTaps % 0x4U;

125#else

128 tapCnt = numTaps;

130#endif /* #if defined (ARM_MATH_LOOPUNROLL) */

137 /* Decrement the loop counter */
138 tapCnt--;

141 /* Calc lower part of acc */

143
144 /* Calc upper part of acc */
145 acc_h = (acc >> 32) & 0xffffffff;

147 /* Apply shift for lower part of acc and upper part of acc */
148 acc = (uint32_t) acc_l >> lShift | acc_h << uShift;
149
150 /* Converting the result to 1.15 format and saturate the output */
151 acc = __SSAT(acc, 16U);
152
153 /* Store the result from accumulator into the destination buffer. */
154 *pOut++ = (q15_t) acc;

188 *pb++ = (q15_t) __SSAT((coef), 16);
189

191 tapCnt--;
192 }

195 tapCnt = numTaps % 0x4U;
196

213
214 /* Decrement loop counter */
215 blkCnt--;
216 }
217
218 /* Processing is complete.
219 Now copy the last numTaps - 1 samples to the start of the state buffer.
220 This prepares the state buffer for the next function call. */

223 pStateCurnt = S->pState;
224
225 /* copy data */
226#if defined (ARM_MATH_LOOPUNROLL)
227

229 tapCnt = (numTaps - 1U) >> 2U;

234 write_q15x2_ia (&pStateCurnt, read_q15x2_ia (&pState));

237 tapCnt--;

243#else

246 tapCnt = (numTaps - 1U);
247
248#endif /* #if defined (ARM_MATH_LOOPUNROLL) */

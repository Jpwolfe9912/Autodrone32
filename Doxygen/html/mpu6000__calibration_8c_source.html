40// MPU6000 Calibration
42
43void mpu6000Calibration(void)
44{
45 uint16_t sampleRate = 100;
46 uint16_t numberOfSamples = 2000;
47
48 float accelBias1[3] = { 0.0f, 0.0f, 0.0f };
49 float gyroBias1[3] = { 0.0f, 0.0f, 0.0f };
50 float mpu6000Temperature1 = 0.0f;
51
52 float accelBias2[3] = { 0.0f, 0.0f, 0.0f };
53 float gyroBias2[3] = { 0.0f, 0.0f, 0.0f };
54 float mpu6000Temperature2 = 0.0f;
55
56 eepromConfig.accelScaleFactorMPU[XAXIS] = 0.004788403f; // (1/8192) * 9.8065 (8192 LSB = 1 G)
57 eepromConfig.accelScaleFactorMPU[YAXIS] = 0.004788403f; // (1/8192) * 9.8065 (8192 LSB = 1 G)
58 eepromConfig.accelScaleFactorMPU[ZAXIS] = 0.004788403f; // (1/8192) * 9.8065 (8192 LSB = 1 G)
59
60 uint16_t index;
61
62 mpu6000Calibrating = true;
63
64 printf("\nMPU6000 Calibration:\n");
65
67 // Get samples at temperature1
69
70 printf("\nBegin 1st MPU6000 Measurements...\n");
71 for (index = 0; index < numberOfSamples; index++)
72 {
73 readMPU6000();
74
75 rawAccel[ZAXIS].value = rawAccel[ZAXIS].value - 2048;
76
77 accelBias1[XAXIS] += rawAccel[XAXIS].value;
78 accelBias1[YAXIS] += rawAccel[YAXIS].value;
79 accelBias1[ZAXIS] += rawAccel[ZAXIS].value;
80 gyroBias1[ROLL ] += rawGyro[ROLL ].value;
81 gyroBias1[PITCH] += rawGyro[PITCH].value;
82 gyroBias1[YAW ] += rawGyro[YAW ].value;
83 mpu6000Temperature1 += (float) (rawMPU6000Temperature.value) / 340.0f + 35.0f;
84
85 delayMicroseconds(sampleRate);
86 }
87
88 accelBias1[XAXIS] /= (float) numberOfSamples;
89 accelBias1[YAXIS] /= (float) numberOfSamples;
90 accelBias1[ZAXIS] /= (float) numberOfSamples;
91 gyroBias1[ROLL ] /= (float) numberOfSamples;
92 gyroBias1[PITCH] /= (float) numberOfSamples;
93 gyroBias1[YAW ] /= (float) numberOfSamples;
94 mpu6000Temperature1 /= (float) numberOfSamples;
95
96 printf("\nGyro Temperature Reading: %6.2f", mpu6000Temperature1);
97
98 printf("\n\nEnd 1st MPU6000 Measurements\n");
99
101 // Time delay for temperature
102 // Stabilization
104
105 printf("\nWaiting for 10 seconds for MPU6000 temp to rise...\n");
106 delay(3000); // Number of mSec in 10 minutes
107
109 // Get samples at temperature2
111
112 printf("\nBegin 2nd MPU6000 Measurements...\n");
113 for (index = 0; index < numberOfSamples; index++)
114 {
115 readMPU6000();
116
117 rawAccel[ZAXIS].value = rawAccel[ZAXIS].value - 2048;
118
119 accelBias2[XAXIS] += rawAccel[XAXIS].value;
120 accelBias2[YAXIS] += rawAccel[YAXIS].value;
121 accelBias2[ZAXIS] += rawAccel[ZAXIS].value;
122 gyroBias2[ROLL ] += rawGyro[ROLL ].value;
123 gyroBias2[PITCH] += rawGyro[PITCH].value;
124 gyroBias2[YAW ] += rawGyro[YAW ].value;
125 mpu6000Temperature2 += (float)(rawMPU6000Temperature.value) / 340.0f + 35.0f;
126
127 delayMicroseconds(sampleRate);
128 }
129
130 accelBias2[XAXIS] /= (float) numberOfSamples;
131 accelBias2[YAXIS] /= (float) numberOfSamples;
132 accelBias2[ZAXIS] /= (float) numberOfSamples;
133 gyroBias2[ROLL ] /= (float) numberOfSamples;
134 gyroBias2[PITCH] /= (float) numberOfSamples;
135 gyroBias2[YAW ] /= (float) numberOfSamples;
136 mpu6000Temperature2 /= (float) numberOfSamples;
137
138 printf("\nGyro Temperature Reading: %6.2f", mpu6000Temperature2);
139
140 printf("\n\nEnd 2nd MPU6000 Measurements\n");
141
143
144 eepromConfig.accelTCBiasSlope[XAXIS] = (accelBias2[XAXIS] - accelBias1[XAXIS]) / (mpu6000Temperature2 - mpu6000Temperature1);
145 eepromConfig.accelTCBiasSlope[YAXIS] = (accelBias2[YAXIS] - accelBias1[YAXIS]) / (mpu6000Temperature2 - mpu6000Temperature1);
146 eepromConfig.accelTCBiasSlope[ZAXIS] = (accelBias2[ZAXIS] - accelBias1[ZAXIS]) / (mpu6000Temperature2 - mpu6000Temperature1);
147
148 eepromConfig.accelTCBiasIntercept[XAXIS] = accelBias2[XAXIS] - eepromConfig.accelTCBiasSlope[XAXIS] * mpu6000Temperature2;
149 eepromConfig.accelTCBiasIntercept[YAXIS] = accelBias2[YAXIS] - eepromConfig.accelTCBiasSlope[YAXIS] * mpu6000Temperature2;
150 eepromConfig.accelTCBiasIntercept[ZAXIS] = accelBias2[ZAXIS] - eepromConfig.accelTCBiasSlope[ZAXIS] * mpu6000Temperature2;
151
152 eepromConfig.gyroTCBiasSlope[ROLL ] = (gyroBias2[ROLL ] - gyroBias1[ROLL ]) / (mpu6000Temperature2 - mpu6000Temperature1);
153 eepromConfig.gyroTCBiasSlope[PITCH] = (gyroBias2[PITCH] - gyroBias1[PITCH]) / (mpu6000Temperature2 - mpu6000Temperature1);
154 eepromConfig.gyroTCBiasSlope[YAW ] = (gyroBias2[YAW ] - gyroBias1[YAW ]) / (mpu6000Temperature2 - mpu6000Temperature1);
155
156 eepromConfig.gyroTCBiasIntercept[ROLL ] = gyroBias2[ROLL ] - eepromConfig.gyroTCBiasSlope[ROLL ] * mpu6000Temperature2;
157 eepromConfig.gyroTCBiasIntercept[PITCH] = gyroBias2[PITCH] - eepromConfig.gyroTCBiasSlope[PITCH] * mpu6000Temperature2;
158 eepromConfig.gyroTCBiasIntercept[YAW ] = gyroBias2[YAW ] - eepromConfig.gyroTCBiasSlope[YAW ] * mpu6000Temperature2;
159
161
162 printf("\nMPU6000 Calibration Complete.\n");
163
164 mpu6000Calibrating = false;
165}

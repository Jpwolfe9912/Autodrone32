65 sbrk
66 Increase program data space.
67 Malloc and related functions depend on this
68 */
69
70caddr_t _sbrk(int incr)
71{
72 extern char _ebss; // Defined by the linker
73 static char *heap_end;
74 char *prev_heap_end;
75
76 char * stack;
77
78 if (heap_end == 0)
79 heap_end = &_ebss;
80
81 prev_heap_end = heap_end;
82
83 stack = (char*) __get_MSP();
84 if (heap_end + incr > stack)
85 {
86 errno = ENOMEM;
87 return (caddr_t) -1;
88 }
89
90 heap_end += incr;
91 return (caddr_t) prev_heap_end;
92}

109// String to Float Conversion
111
112// Simple and fast atof (ascii to float) function.
113//
114// - Executes about 5x faster than standard MSCRT library atof().
115// - An attractive alternative if the number of calls is in the millions.
116// - Assumes input is a proper integer, fraction, or scientific format.
117// - Matches library atof() to 15 digits (except at extreme exponents).
118// - Follows atof() precedent of essentially no error checking.
119//
120// 09-May-2009 Tom Van Baak (tvb) www.LeapSecond.com
121//
122
123#define white_space(c) ((c) == ' ' || (c) == '\t')
124#define valid_digit(c) ((c) >= '0' && (c) <= '9')
125
126float stringToFloat(const char *p)
127{
128 int frac = 0;
129 double sign, value, scale;
130
131 // Skip leading white space, if any.
132
133 while (white_space(*p) ) {
134 p += 1;
135 }
136
137 // Get sign, if any.
138
139 sign = 1.0;
140 if (*p == '-') {
141 sign = -1.0;
142 p += 1;
143
144 } else if (*p == '+') {
145 p += 1;
146 }
147
148 // Get digits before decimal point or exponent, if any.
149
150 value = 0.0;
151 while (valid_digit(*p)) {
152 value = value * 10.0 + (*p - '0');
153 p += 1;
154 }
155
156 // Get digits after decimal point, if any.
157
158 if (*p == '.') {
159 double pow10 = 10.0;
160 p += 1;
161
162 while (valid_digit(*p)) {
163 value += (*p - '0') / pow10;
164 pow10 *= 10.0;
165 p += 1;
166 }
167 }
168
169 // Handle exponent, if any.
170
171 scale = 1.0;
172 if ((*p == 'e') || (*p == 'E')) {
173 unsigned int expon;
174 p += 1;
175
176 // Get sign of exponent, if any.
177
178 frac = 0;
179 if (*p == '-') {
180 frac = 1;
181 p += 1;
182
183 } else if (*p == '+') {
184 p += 1;
185 }
186
187 // Get digits of exponent, if any.
188
189 expon = 0;
190 while (valid_digit(*p)) {
191 expon = expon * 10 + (*p - '0');
192 p += 1;
193 }
194 if (expon > 308) expon = 308;
195
196 // Calculate scaling factor.
197
198 while (expon >= 50) { scale *= 1E50; expon -= 50; }
199 while (expon >= 8) { scale *= 1E8; expon -= 8; }
200 while (expon > 0) { scale *= 10.0; expon -= 1; }
201 }
202
203 // Return signed and scaled floating point result.
204
205 return sign * (frac ? (value / scale) : (value * scale));
206}

60
61void computeAxisCommands(float dt)
62{
63 float error;
64 float tempAttCompensation;
65
66 if (mode == FLIGHT)
67 {
68 attCmd[ROLL ] = rxCommand[ROLL ] * eepromConfig.attitudeScaling;
69 attCmd[PITCH] = rxCommand[PITCH] * eepromConfig.attitudeScaling;
70 rateCmd[YAW] = rxCommand[YAW ] * eepromConfig.yawRateScaling;
71
72 error = standardRadianFormat(attCmd[ROLL] - sensors.attitude500Hz[ROLL]);
73 attPID[ROLL] = updatePID(error, dt, pidReset, &eepromConfig.PID[ROLL_ATT_PID ]);
74
75 error = standardRadianFormat(attCmd[PITCH] + sensors.attitude500Hz[PITCH]);
76 attPID[PITCH] = updatePID(error, dt, pidReset, &eepromConfig.PID[PITCH_ATT_PID]);
77
78 error = rateCmd[YAW] - sensors.gyro500Hz[YAW];
79 ratePID[YAW] = updatePID(error, dt, pidReset, &eepromConfig.PID[YAW_RATE_PID ]);
80
82
83 throttleCmd = rxCommand[THROTTLE];
84 }

87}

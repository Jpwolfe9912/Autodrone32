173HAL_StatusTypeDef USB_SetTurnaroundTime(USB_OTG_GlobalTypeDef *USBx,
174 uint32_t hclk, uint8_t speed)
175{
176 uint32_t UsbTrd;
177
178 /* The USBTRD is configured according to the tables below, depending on AHB frequency
179 used by application. In the low AHB frequency range it is used to stretch enough the USB response
180 time to IN tokens, the USB turnaround time, so to compensate for the longer AHB read access
181 latency to the Data FIFO */
182 if (speed == USBD_FS_SPEED)
183 {
184 if ((hclk >= 14200000U) && (hclk < 15000000U))
185 {
186 /* hclk Clock Range between 14.2-15 MHz */
187 UsbTrd = 0xFU;
188 }
189 else if ((hclk >= 15000000U) && (hclk < 16000000U))
190 {
191 /* hclk Clock Range between 15-16 MHz */
192 UsbTrd = 0xEU;
193 }
194 else if ((hclk >= 16000000U) && (hclk < 17200000U))
195 {
196 /* hclk Clock Range between 16-17.2 MHz */
197 UsbTrd = 0xDU;
198 }
199 else if ((hclk >= 17200000U) && (hclk < 18500000U))
200 {
201 /* hclk Clock Range between 17.2-18.5 MHz */
202 UsbTrd = 0xCU;
203 }
204 else if ((hclk >= 18500000U) && (hclk < 20000000U))
205 {
206 /* hclk Clock Range between 18.5-20 MHz */
207 UsbTrd = 0xBU;
208 }
209 else if ((hclk >= 20000000U) && (hclk < 21800000U))
210 {
211 /* hclk Clock Range between 20-21.8 MHz */
212 UsbTrd = 0xAU;
213 }
214 else if ((hclk >= 21800000U) && (hclk < 24000000U))
215 {
216 /* hclk Clock Range between 21.8-24 MHz */
217 UsbTrd = 0x9U;
218 }
219 else if ((hclk >= 24000000U) && (hclk < 27700000U))
220 {
221 /* hclk Clock Range between 24-27.7 MHz */
222 UsbTrd = 0x8U;
223 }
224 else if ((hclk >= 27700000U) && (hclk < 32000000U))
225 {
226 /* hclk Clock Range between 27.7-32 MHz */
227 UsbTrd = 0x7U;
228 }
229 else /* if(hclk >= 32000000) */
230 {
231 /* hclk Clock Range between 32-200 MHz */
232 UsbTrd = 0x6U;
233 }
234 }
235 else if (speed == USBD_HS_SPEED)
236 {
237 UsbTrd = USBD_HS_TRDT_VALUE;
238 }
239 else
240 {
241 UsbTrd = USBD_DEFAULT_TRDT_VALUE;
242 }
243
244 USBx->GUSBCFG &= ~USB_OTG_GUSBCFG_TRDT;
245 USBx->GUSBCFG |= (uint32_t)((UsbTrd << 10) & USB_OTG_GUSBCFG_TRDT);
246
247 return HAL_OK;
248}

500HAL_StatusTypeDef USB_FlushTxFifo(USB_OTG_GlobalTypeDef *USBx, uint32_t num)
501{
502 __IO uint32_t count = 0U;
503
504 /* Wait for AHB master IDLE state. */
505 do
506 {
507 if (++count > 200000U)
508 {
509 return HAL_TIMEOUT;
510 }
511 } while ((USBx->GRSTCTL & USB_OTG_GRSTCTL_AHBIDL) == 0U);
512
513 /* Flush TX Fifo */
514 count = 0U;
515 USBx->GRSTCTL = (USB_OTG_GRSTCTL_TXFFLSH | (num << 6));
516
517 do
518 {
519 if (++count > 200000U)
520 {
521 return HAL_TIMEOUT;
522 }
523 } while ((USBx->GRSTCTL & USB_OTG_GRSTCTL_TXFFLSH) == USB_OTG_GRSTCTL_TXFFLSH);
524
525 return HAL_OK;
526}

783HAL_StatusTypeDef USB_EPStartXfer(USB_OTG_GlobalTypeDef *USBx, USB_OTG_EPTypeDef *ep, uint8_t dma)
784{
785 uint32_t USBx_BASE = (uint32_t)USBx;
786 uint32_t epnum = (uint32_t)ep->num;
787 uint16_t pktcnt;
788
789 /* IN endpoint */
790 if (ep->is_in == 1U)
791 {
792 /* Zero Length Packet? */
793 if (ep->xfer_len == 0U)
794 {
795 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_PKTCNT);
796 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_PKTCNT & (1U << 19));
797 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_XFRSIZ);
798 }
799 else
800 {
801 /* Program the transfer size and packet count
802 * as follows: xfersize = N * maxpacket +
803 * short_packet pktcnt = N + (short_packet
804 * exist ? 1 : 0)
805 */
806 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_XFRSIZ);
807 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_PKTCNT);
808 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_PKTCNT &
809 (((ep->xfer_len + ep->maxpacket - 1U) / ep->maxpacket) << 19));
810
811 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_XFRSIZ & ep->xfer_len);
812
813 if (ep->type == EP_TYPE_ISOC)
814 {
815 USBx_INEP(epnum)->DIEPTSIZ &= ~(USB_OTG_DIEPTSIZ_MULCNT);
816 USBx_INEP(epnum)->DIEPTSIZ |= (USB_OTG_DIEPTSIZ_MULCNT & (1U << 29));
817 }
818 }
819
820 if (dma == 1U)
821 {
822 if ((uint32_t)ep->dma_addr != 0U)
823 {
824 USBx_INEP(epnum)->DIEPDMA = (uint32_t)(ep->dma_addr);
825 }
826
827 if (ep->type == EP_TYPE_ISOC)
828 {
829 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
830 {
831 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SODDFRM;
832 }
833 else
834 {
835 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SD0PID_SEVNFRM;
836 }
837 }
838
839 /* EP enable, IN data in FIFO */
840 USBx_INEP(epnum)->DIEPCTL |= (USB_OTG_DIEPCTL_CNAK | USB_OTG_DIEPCTL_EPENA);
841 }
842 else
843 {
844 /* EP enable, IN data in FIFO */
845 USBx_INEP(epnum)->DIEPCTL |= (USB_OTG_DIEPCTL_CNAK | USB_OTG_DIEPCTL_EPENA);
846
847 if (ep->type != EP_TYPE_ISOC)
848 {
849 /* Enable the Tx FIFO Empty Interrupt for this EP */
850 if (ep->xfer_len > 0U)
851 {
852 USBx_DEVICE->DIEPEMPMSK |= 1UL << (ep->num & EP_ADDR_MSK);
853 }
854 }
855 else
856 {
857 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
858 {
859 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SODDFRM;
860 }
861 else
862 {
863 USBx_INEP(epnum)->DIEPCTL |= USB_OTG_DIEPCTL_SD0PID_SEVNFRM;
864 }
865
866 (void)USB_WritePacket(USBx, ep->xfer_buff, ep->num, (uint16_t)ep->xfer_len, dma);
867 }
868 }
869 }
870 else /* OUT endpoint */
871 {
872 /* Program the transfer size and packet count as follows:
873 * pktcnt = N
874 * xfersize = N * maxpacket
875 */
876 USBx_OUTEP(epnum)->DOEPTSIZ &= ~(USB_OTG_DOEPTSIZ_XFRSIZ);
877 USBx_OUTEP(epnum)->DOEPTSIZ &= ~(USB_OTG_DOEPTSIZ_PKTCNT);
878
879 if (ep->xfer_len == 0U)
880 {
881 USBx_OUTEP(epnum)->DOEPTSIZ |= (USB_OTG_DOEPTSIZ_XFRSIZ & ep->maxpacket);
882 USBx_OUTEP(epnum)->DOEPTSIZ |= (USB_OTG_DOEPTSIZ_PKTCNT & (1U << 19));
883 }
884 else
885 {
886 pktcnt = (uint16_t)((ep->xfer_len + ep->maxpacket - 1U) / ep->maxpacket);
887 USBx_OUTEP(epnum)->DOEPTSIZ |= USB_OTG_DOEPTSIZ_PKTCNT & ((uint32_t)pktcnt << 19);
888 USBx_OUTEP(epnum)->DOEPTSIZ |= USB_OTG_DOEPTSIZ_XFRSIZ & (ep->maxpacket * pktcnt);
889 }
890
891 if (dma == 1U)
892 {
893 if ((uint32_t)ep->xfer_buff != 0U)
894 {
895 USBx_OUTEP(epnum)->DOEPDMA = (uint32_t)(ep->xfer_buff);
896 }
897 }
898
899 if (ep->type == EP_TYPE_ISOC)
900 {
901 if ((USBx_DEVICE->DSTS & (1U << 8)) == 0U)
902 {
903 USBx_OUTEP(epnum)->DOEPCTL |= USB_OTG_DOEPCTL_SODDFRM;
904 }
905 else
906 {
907 USBx_OUTEP(epnum)->DOEPCTL |= USB_OTG_DOEPCTL_SD0PID_SEVNFRM;
908 }
909 }
910 /* EP enable */
911 USBx_OUTEP(epnum)->DOEPCTL |= (USB_OTG_DOEPCTL_CNAK | USB_OTG_DOEPCTL_EPENA);
912 }
913
914 return HAL_OK;
915}

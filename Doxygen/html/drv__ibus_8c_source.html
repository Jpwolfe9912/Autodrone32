100static ibusStatus_e
101ibus_process_frame(void)
102{
103 uint8_t b;
104 if(lwrb_read(&rxRingBuf, &b, 1) == 1){
105 status = IBUS_BUSY;
106 switch(state){
107 case IBUS_STATE_LENGTH: { // length byte
108 if(b == 0x20){
109 frameLength = b;
110 framePos = 0;
111 ++state;
112 }
113 else
114 status = IBUS_ERROR;
115 break;
116 }
117 case IBUS_STATE_TYPE: {
118 cmd = b;
119 if(cmd == 0x40){
120 dev_id = cmd;
121 ++state;
122 }
123 else
124 status = IBUS_ERROR;
125 break;
126 }
127 case IBUS_STATE_PAYLOAD: { // payload bytes
128 ibusPayload[framePos++] = b;
129 --frameLength;
130 if(framePos == PAYLOAD_SIZE){
131 ibus_update(ibusPayload);
132 ++state;
133 framePos = 0;
134 }
135 break;
136 }
137 case IBUS_STATE_CRC: {
138 if(framePos == 1){
139 if(ibusFrameCRC()){
140 state = 0;
141 status = IBUS_READY;
142 }
143 else{
144 status = IBUS_ERROR;
145 }
146 }
147 else{
148 framePos++;
149 }
150 break;
151 }
152 }
153 }
154 if(status == IBUS_ERROR){
155 ibus_error_timeout();
156 lwrb_reset(&rxRingBuf);
157 memset(rxBuf, 0x00, RXBUF_SIZE);
158 }
159 errorTimeOut = 0;
160 return status;
161}

240static void
241usart_rx_check(uint32_t bytesLeft)
242{
243 static size_t old_pos;
244 size_t pos;
245
246 /* Calculate current position in buffer and check for new data available */
247 pos = ARRAY_LEN(rxBuf) - bytesLeft;
248 if (pos != old_pos) { /* Check change in received data */
249 if (pos > old_pos) { /* Current position is over previous one */
250 /*
251 * Processing is done in "linear" mode.
252 *
253 * Application processing is fast with single data block,
254 * length is simply calculated by subtracting pointers
255 *
256 * [ 0 ]
257 * [ 1 ] <- old_pos |------------------------------------|
258 * [ 2 ] | |
259 * [ 3 ] | Single block (len = pos - old_pos) |
260 * [ 4 ] | |
261 * [ 5 ] |------------------------------------|
262 * [ 6 ] <- pos
263 * [ 7 ]
264 * [ N - 1 ]
265 */
266 usart_process_data(&rxBuf[old_pos], pos - old_pos);
267 }
268 else {
269 /*
270 * Processing is done in "overflow" mode..
271 *
272 * Application must process data twice,
273 * since there are 2 linear memory blocks to handle
274 *
275 * [ 0 ] |---------------------------------|
276 * [ 1 ] | Second block (len = pos) |
277 * [ 2 ] |---------------------------------|
278 * [ 3 ] <- pos
279 * [ 4 ] <- old_pos |---------------------------------|
280 * [ 5 ] | |
281 * [ 6 ] | First block (len = N - old_pos) |
282 * [ 7 ] | |
283 * [ N - 1 ] |---------------------------------|
284 */
285 usart_process_data(&rxBuf[old_pos], ARRAY_LEN(rxBuf) - old_pos);
286 if (pos > 0) {
287 usart_process_data(&rxBuf[0], pos);
288 }
289 }
290 old_pos = pos; /* Save current position as old for next transfers */
291 }
292}

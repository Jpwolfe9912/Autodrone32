101void arm_biquad_cascade_df2T_compute_coefs_f32(
102 arm_biquad_cascade_df2T_instance_f32 * S,
103 uint8_t numStages,
104 float32_t * pCoeffs)
105{
106 uint8_t cnt;
107 float32_t *pDstCoeffs;
108 float32_t b0[4],b1[4],b2[4],a1[4],a2[4];
109
110 pDstCoeffs = (float32_t*)S->pCoeffs;
111
112 cnt = numStages >> 2;
113 while(cnt > 0)
114 {
115 for(int i=0;i<4;i++)
116 {
117 b0[i] = pCoeffs[0];
118 b1[i] = pCoeffs[1];
119 b2[i] = pCoeffs[2];
120 a1[i] = pCoeffs[3];
121 a2[i] = pCoeffs[4];
122 pCoeffs += 5;
123 }
124
125 /* Vec 1 */
126 *pDstCoeffs++ = 0;
127 *pDstCoeffs++ = b0[1];
128 *pDstCoeffs++ = b0[2];
129 *pDstCoeffs++ = b0[3];
130
131 /* Vec 2 */
132 *pDstCoeffs++ = 0;
133 *pDstCoeffs++ = 0;
134 *pDstCoeffs++ = b0[1] * b0[2];
135 *pDstCoeffs++ = b0[2] * b0[3];
136
137 /* Vec 3 */
138 *pDstCoeffs++ = 0;
139 *pDstCoeffs++ = 0;
140 *pDstCoeffs++ = 0;
141 *pDstCoeffs++ = b0[1] * b0[2] * b0[3];
142
143 /* Vec 4 */
144 *pDstCoeffs++ = b0[0];
145 *pDstCoeffs++ = b0[0] * b0[1];
146 *pDstCoeffs++ = b0[0] * b0[1] * b0[2];
147 *pDstCoeffs++ = b0[0] * b0[1] * b0[2] * b0[3];
148
149 /* Vec 5 */
150 *pDstCoeffs++ = b1[0];
151 *pDstCoeffs++ = b1[1];
152 *pDstCoeffs++ = b1[2];
153 *pDstCoeffs++ = b1[3];
154
155 /* Vec 6 */
156 *pDstCoeffs++ = b2[0];
157 *pDstCoeffs++ = b2[1];
158 *pDstCoeffs++ = b2[2];
159 *pDstCoeffs++ = b2[3];
160
161 /* Vec 7 */
162 *pDstCoeffs++ = a1[0];
163 *pDstCoeffs++ = a1[1];
164 *pDstCoeffs++ = a1[2];
165 *pDstCoeffs++ = a1[3];
166
167 /* Vec 8 */
168 *pDstCoeffs++ = a2[0];
169 *pDstCoeffs++ = a2[1];
170 *pDstCoeffs++ = a2[2];
171 *pDstCoeffs++ = a2[3];
172
173 cnt--;
174 }
175
176 cnt = numStages & 0x3;
177 while(cnt > 0)
178 {
179 *pDstCoeffs++ = *pCoeffs++;
180 *pDstCoeffs++ = *pCoeffs++;
181 *pDstCoeffs++ = *pCoeffs++;
182 *pDstCoeffs++ = *pCoeffs++;
183 *pDstCoeffs++ = *pCoeffs++;
184 cnt--;
185 }
186
187}

190void arm_biquad_cascade_df2T_init_f32(
191 arm_biquad_cascade_df2T_instance_f32 * S,
192 uint8_t numStages,
193 const float32_t * pCoeffs,
194 float32_t * pState)
195{
196 /* Assign filter stages */
197 S->numStages = numStages;
198
199 /* Assign coefficient pointer */
200 S->pCoeffs = pCoeffs;
201
202 /* Clear state buffer and size is always 2 * numStages */
203 memset(pState, 0, (2U * (uint32_t) numStages) * sizeof(float32_t));
204
205 /* Assign state pointer */
206 S->pState = pState;
207}

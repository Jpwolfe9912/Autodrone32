309static void
310dshot_prepare_dmabuffer(uint32_t *motor_dmabuffer, uint16_t value)
311{
312 uint16_t packet;
313 packet = dshot_prepare_packet(value);
314
315 for(int i = 0; i < 16; i++)
316 {
317 motor_dmabuffer[i] = (packet & 0x8000) ? MOTOR_BIT_1 : MOTOR_BIT_0;
318 packet <<= 1;
319 }
320
321 motor_dmabuffer[16] = 0;
322 motor_dmabuffer[17] = 0;
323
324}

334static uint16_t
335dshot_prepare_packet(uint16_t value)
336{
337 uint16_t packet;
338 bool dshot_telemetry = false;
339
340 packet = (value << 1) | (dshot_telemetry ? 1 : 0);
341
342 // compute checksum
343 unsigned csum = 0;
344 unsigned csum_data = packet;
345
346 for(int i = 0; i < 3; i++)
347 {
348 csum ^= csum_data; // xor data by nibbles
349 csum_data >>= 4;
350 }
351
352 csum &= 0xf;
353 packet = (packet << 4) | csum;
354
355 return packet;
356}

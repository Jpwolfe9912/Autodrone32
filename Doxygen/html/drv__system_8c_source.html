161uint32_t
162micros(void)
163{
164 register uint32_t oldCycle, cycle, timeMs;
165
166 do
167 {
168 timeMs = __LDREXW(&sysTickUptime);
169 cycle = DWT->CYCCNT;
170 oldCycle = sysTickCycleCounter;
171 }
172 while ( __STREXW( timeMs , &sysTickUptime ) );
173
174 return (timeMs * 1000) + (cycle - oldCycle) / usTicks;
175}

200void delayMicroseconds(uint32_t us)
201{
202 uint32_t elapsed = 0;
203 uint32_t lastCount = DWT->CYCCNT;
204
205 for (;;) {
206 register uint32_t current_count = DWT->CYCCNT;
207 uint32_t elapsed_us;
208
209 // measure the time elapsed since the last time we checked
210 elapsed += current_count - lastCount;
211 lastCount = current_count;
212
213 // convert to microseconds
214 elapsed_us = elapsed / usTicks;
215 if (elapsed_us >= us)
216 break;
217
218 // reduce the delay by the elapsed time
219 us -= elapsed_us;
220
221 // keep fractional microseconds for the next iteration
222 elapsed %= usTicks;
223 }
224}

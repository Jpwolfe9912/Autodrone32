189micros(void)
190{
191 register uint32_t oldCycle, cycle, timeMs;
192
193 do
194 {
195 timeMs = __LDREXW(&sysTickUptime);
196 cycle = DWT->CYCCNT;
197 oldCycle = sysTickCycleCounter;
198 }
199 while ( __STREXW( timeMs , &sysTickUptime ) );
200
201 return (timeMs * 1000) + (cycle - oldCycle) / usTicks;

227void delayMicroseconds(uint32_t us)
228{
229 uint32_t elapsed = 0;
230 uint32_t lastCount = DWT->CYCCNT;
231
232 for (;;) {
233 register uint32_t current_count = DWT->CYCCNT;
234 uint32_t elapsed_us;
235
236 // measure the time elapsed since the last time we checked
237 elapsed += current_count - lastCount;
238 lastCount = current_count;
239
240 // convert to microseconds
241 elapsed_us = elapsed / usTicks;
242 if (elapsed_us >= us)
243 break;
244
245 // reduce the delay by the elapsed time
246 us -= elapsed_us;
247
248 // keep fractional microseconds for the next iteration
249 elapsed %= usTicks;
250 }
251}
